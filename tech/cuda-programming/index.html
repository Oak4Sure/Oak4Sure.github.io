Brief introduction to Cuda Programming using C++. 使用C++进行Cuda编程，进行GPU并行化。

#Cuda编程

#Cuda编程的移植

将计算密集型函数移植到GPU上以实现并行化，而其他顺序执行的代码仍然在CPU上执行

#简单处理流程

  1. 分配host内存,并进行数据初始化
  2. 分配device内存,并从host将数据copy到device上
    cudaMalloc & cudaMemcpy
  3. 调用Cuda的核函数在device上完成指定的运算
    <<<Blocks, Threads>>>
  4. 将device上的运算结果copy到host上
  5. 释放device和host上分配的内存
    cudaFree & delete[] or free[]

__global__ void add(int n, float *x, float *y) {
    int index = threadIdx.x;
    int stride = blockDim.x;
    for (int i = index; i < n; i += stride) {
        y[i] = x[i] + y[i];
    }
}

#managed

#核函数限制条件

  • 只能访问设备内存
  • 必须返回void
  • 不支持可变参数的函数int func(int n_args, ...)
  • 参数不可为引用类型（无法访问主机内存）
  • 不支持静态变量（存在主机内存的数据段或者BSS段）

#主机内存和设备内存

他们是分隔的两个部分

  • 设备指针指向GPU内存
    • 通常不会传给主机代码
    • 通常主机代码不会引用
  • 主机指针指向CPU内存
    • 通常不会传给设备代码
    • 通常设备代码不会引用

#Cuda 使用一个Block

1
2
3
4
5

__global__ void vector_add(int *a, int* b, int* c){
    int tid = threadIdx.x;  // 获取线程编号
    c[tid] = a[tid] + b[tid];   // 每个线程完成一个元素的加和
}
vector_add<<< 1, n >>>(a, b, c);

#存在问题

  • a, b, c为主内存地址，GPU无法访问
  • block中有最大线程数限制：n必须不大于1024
  • 同一个block只在一个SM上执行：没有充分利用GPU计算资源

  • 释放：cudaFree

int *a_d, *b_d, *c_d; //_d常用来表明设备内存
int n_bytes = sizeof(int)*n;
cudaMalloc((void**)&a_d, sizeof(int)*n);    // 创建device内存

... //same for b and cudaMalloc for c
vector_add<<< 1, n >>>(a_d, b_d, c_d);  // 调用核函数执行GPU代码
cudaDeviceSynchronize();    // 等待所有加和操作完成，CPU再把结果拷贝回来
cudaMemcpy(c_h, c_d, n_bytes, cudaMemcpyDeviceToHost);
cudaFree(a_d);  // 释放GPU内存
... //same for b and c

#Cuda使用多个Block

  • 每个Block使用m个thread
  • 确定thread的全局编号
1
2
3
4
5

__global__ void vector_add(int *a, int* b, int* c){
    int tid = blockDim.x * blockIdx.x + threadIdx.x;
    c[tid] = a[tid] + b[tid];
}
vector_add<<< n/m, m>>>(a, b, c);

#存在问题

  • n无法被m整除
  • 需对n/m向上取整，才是block的数量
  • 需判断tid是否超过n
  • 一般每个block里面的线程数目m是确定的

int tid = blockDim.x * blockIdx.x + threadIdx.x;
    if (tid < n){   // 判断tid是否越界

int divup(int n, int m){
    return ((n%m)?(n/m+1):(n/m));   // 向上取整
}
vector_add<<< divup(n,m), m>>>(a, b, c, n);

#Cuda VS OpenMP

  • Cuda
    • 由核函数指明并行代码
    • 主机代码调用核函数产生设备线程
    • 用户决定每个线程处理的任务
    • 异步执行，同步需要cudaDeviceSynchronize()
  • OpenMP
    • 由预处理指令与{}指明并行区域
    • 主线程产生从线程
    • 由调度算法将任务分配到线程上
    • 默认在并行区域结束的时候同步
